import React, { useState, useEffect } from "react";
import Sidebar from "./components/Sidebar.jsx";
import ChatWindow from "./components/ChatWindow.jsx";
import PersonaSelector from "./components/PersonaSelector.jsx";
import ToolsPanel from "./components/ToolsPanel.jsx";
import api from "./api";

export default function App() {
  const [projects, setProjects] = useState([]);
  const [activeProject, setActiveProject] = useState(null);
  const [messages, setMessages] = useState([]);
  const [activePersona, setActivePersona] = useState("developer");

  useEffect(() => {
    api.getProjects().then(setProjects);
  }, []);

  const sendMessage = async (content) => {
    const result = await api.sendMessage(activeProject, activePersona, content);
    setMessages((m) => [...m, result.user, result.ai]);
  };

  return (
    <div className="flex h-screen">
      <Sidebar
        projects={projects}
        active={activeProject}
        onSelect={setActiveProject}
      />

      <div className="flex flex-col flex-1">
        <div className="border-b border-gray-700 p-2 bg-gray-800">
          <PersonaSelector value={activePersona} onChange={setActivePersona} />
        </div>

        <ChatWindow messages={messages} onSend={sendMessage} />

        <ToolsPanel />
      </div>
    </div>
  );
}

ðŸ“„ FILE: frontend/src/api/index.js
import { client } from "./client";

export default {
  async getProjects() {
    const res = await client.get("/projects");
    return res.data.projects;
  },

  async sendMessage(projectId, persona, content) {
    const res = await client.post(`/projects/${projectId}/chat`, {
      persona,
      content
    });
    return res.data;
  },

  async runTool(toolName, args = {}) {
    const res = await client.post("/tools/run", { toolName, args });
    return res.data;
  },

  async health() {
    return client.get("/health");
  }
};

ðŸ“„ FILE: frontend/src/components/ChatWindow.jsx
import React, { useState } from "react";
import MessageBubble from "./MessageBubble.jsx";

export default function ChatWindow({ messages, onSend }) {
  const [text, setText] = useState("");

  const submit = (e) => {
    e.preventDefault();
    if (text.trim()) {
      onSend(text);
      setText("");
    }
  };

  return (
    <div className="flex flex-col flex-1 bg-gray-800">
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {messages.map((m, i) => (
          <MessageBubble key={i} role={m.role} content={m.content} />
        ))}
      </div>

      <form onSubmit={submit} className="p-3 border-t border-gray-700">
        <input
          className="w-full p-3 rounded bg-gray-900 text-white"
          value={text}
          onChange={(e) => setText(e.target.value)}
          placeholder="Messageâ€¦"
        />
      </form>
    </div>
  );
}
