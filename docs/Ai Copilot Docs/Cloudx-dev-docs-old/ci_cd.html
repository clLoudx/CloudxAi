3. Build + Test CI (ci.yml)

This job runs on every push and pull request.

name: CI - Build & Test

on:
  push:
    branches: [ main, dev ]
  pull_request:
    branches: [ main, dev ]

jobs:
  tests:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v4

    - name: Setup Python
      uses: actions/setup-python@v5
      with: { python-version: "3.11" }

    - name: Install dependencies
      run: |
        pip install -r requirements.txt

    - name: Run unit tests (PyTest)
      run: |
        pytest --maxfail=1 --disable-warnings -q

    - name: Archive Test Results
      uses: actions/upload-artifact@v4
      with:
        name: test-results
        path: pytest_full_output.txt
✔ Fails immediately on first test failure.

4. Build ZIP + Release Artifact (build-zip-and-test.yml)

name: Build Release ZIP

on:
  workflow_dispatch:
  push:
    tags:
      - "v*"

jobs:
  build-zip:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Install Python
      uses: actions/setup-python@v5
      with: { python-version: "3.11" }

    - name: Install deps
      run: pip install -r requirements.txt

    - name: Build ZIP
      run: bash build_release_zip.sh

    - name: Upload ZIP artifact
      uses: actions/upload-artifact@v4
      with:
        name: aiagent-release
        path: aiagent_v*.zip

5. Docker CI (ci-build-and-push.yml)

name: Build & Push Docker Images

on:
  push:
    branches: [ main ]

jobs:
  docker:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Login to Registry
      uses: docker/login-action@v3
      with:
        registry: ghcr.io
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}

    - name: Build images
      run: |
        docker build -t ghcr.io/${{ github.repository }}/aiagent:latest .
        docker build -t ghcr.io/${{ github.repository }}/aiagent-worker:latest -f worker/Dockerfile .

    - name: Push images
      run: |
        docker push ghcr.io/${{ github.repository }}/aiagent:latest
        docker push ghcr.io/${{ github.repository }}/aiagent-worker:latest

6. Helm Deploy Pipeline (helm-deploy.yml)

name: Deploy to Kubernetes

on:
  workflow_dispatch:
  push:
    branches: [ production ]

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Setup Kubectl
      uses: azure/setup-kubectl@v4

    - name: Configure KubeConfig
      run: |
        mkdir -p ~/.kube
        echo "${{ secrets.KUBECONFIG_PROD }}" > ~/.kube/config

    - name: Helm Install/Upgrade
      run: |
        helm upgrade --install aiagent devops/helm/aiagent \
            --namespace aiagent \
            --values devops/helm/aiagent/values-production.yaml

    - name: Post-deploy smoke tests
      run: |
        bash devops/tools/post_deploy_smoke.sh aiagent aiagent-web /healthz 8000
✔ Helm rollback is automatically triggered on failure.
